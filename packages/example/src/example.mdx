# ReactLiveProps Basic Example with Explicit Knobs

import { PropsTable, Playground } from 'docz'
import { ReactLiveProps, text } from 'react-live-props'
const jsxControlsList = [
  <button key="1">Button 1</button>,
  <button key="2">Button 2</button>,
  <button key="3">Button 3</button>
]

import ExampleComponent from './ExampleComponent'

<PropsTable of={ExampleComponent} />

<ReactLiveProps>
  <ExampleComponent
    bar={text('Bar Knob Desc', 'bar default value')}
    foo={3}
    baz={boolean('Baz Knob Desc', true)}
    uncontrolledBoolean={true}
    jsxControls={jsxControlsList}
  >
    <span>{text('Child span content text', 'Child Element')}</span>
  </ExampleComponent>
</ReactLiveProps>
