# ReactLiveProps Basic Example

import { PropsTable } from 'docz'
import ExampleComponent from './ExampleComponent'
import { ReactLiveProps } from 'react-live-props'
const jsxControlsList = [
  <button>Button 1</button>,
  <button>Button 2</button>,
  <button>Button 3</button>
]

<PropsTable of={ExampleComponent} />

<ReactLiveProps>
  <ExampleComponent
    bar={text('Bar Knob Desc', 'bar default value')}
    foo={3}
    baz={boolean('Baz Knob Desc', true)}
    uncontrolledBoolean={true}
    jsxControls={jsxControlsList}
  >
    <span>{text('Child span content text', 'Child Element')}</span>
  </ExampleComponent>
</ReactLiveProps>
